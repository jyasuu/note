{{- define "mda.mergeSQL" -}}
{{- $target := .target }}
{{- $source := .source }}
{{- $sourceFilter := .sourceFilter }}
{{- $joinKeys := .joinKeys }}
{{- $matchConditions := .matchConditions }}
{{- $cols := .cols }}
MERGE INTO {{ $target }} t
USING (
  SELECT * FROM {{ $source }}
  {{- if $sourceFilter }}
  WHERE {{ $sourceFilter }}
  {{- end }}
) s
ON
  {{- range $i, $key := $joinKeys }}
  {{- if $i }} AND{{ end }} t.{{ $key }} = s.{{ $key }}
  {{- end }}
WHEN MATCHED AND
  {{- range $i, $cond := $matchConditions }}
  {{- if $i }} AND{{ end }} {{ $cond }}
  {{- end }} THEN
  UPDATE SET
    {{- range $i, $col := $cols }}
    {{- if $i }},{{ end }} t.{{ $col }} = s.{{ $col }}
    {{- end }}
WHEN NOT MATCHED THEN
  INSERT (
    {{- range $i, $key := $joinKeys }}
    {{- if $i }}, {{ end }}{{ $key }}
    {{- end }},
    {{- range $i, $col := $cols }}
    {{- if $i }},{{ end }} {{ $col }}
    {{- end }})
  VALUES (
    {{- range $i, $key := $joinKeys }}
    {{- if $i }}, {{ end }}s.{{ $key }}
    {{- end }},
    {{- range $i, $col := $cols }}
    {{- if $i }},{{ end }} s.{{ $col }}
    {{- end }});

SELECT '{{ $target }}', count(1) from {{ $target }};

{{- end }}
